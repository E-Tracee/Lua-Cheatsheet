-- Chain tweens
local function chainTweens(object, tweenData)
    local currentTween = 1
    
    local function playNext()
        if currentTween <= #tweenData then
            local data = tweenData[currentTween]
            local tween = TweenService:Create(object, data.info, data.goal)
            
            tween.Completed:Connect(function()
                currentTween = currentTween + 1
                playNext()
            end)
            
            tween:Play()
        end
    end
    
    playNext()
end

üèóÔ∏è Object-Oriented Programming

-- Simple class
local Player = {}
Player.__index = Player

function Player.new(name, health)
    local self = setmetatable({}, Player)
    self.name = name
    self.health = health or 100
    self.maxHealth = self.health
    return self
end

function Player:takeDamage(amount)
    self.health = math.max(0, self.health - amount)
    if self.health == 0 then
        self:die()
    end
end

function Player:heal(amount)
    self.health = math.min(self.maxHealth, self.health + amount)
end

function Player:die()
    print(self.name .. " has died!")
end

-- Usage
local player1 = Player.new("John", 150)
player1:takeDamage(50)

Inheritance

-- Warrior class inherits from Player
local Warrior = setmetatable({}, {__index = Player})
Warrior.__index = Warrior

function Warrior.new(name, health, armor)
    local self = Player.new(name, health)
    setmetatable(self, Warrior)
    self.armor = armor or 20
    return self
end

function Warrior:takeDamage(amount)
    local reducedDamage = math.max(1, amount - self.armor)
    Player.takeDamage(self, reducedDamage)
end

function Warrior:charge(target)
    print(self.name .. " charges at " .. target.name .. "!")
    target:takeDamage(30)
end

üîÆ Metatables & Metamethods

-- Basic metatable
local mt = {
    __index = function(table, key)
        return "Key not found: " .. key
    end,
    
    __newindex = function(table, key, value)
        print("Setting " .. key .. " to " .. tostring(value))
        rawset(table, key, value)
    end,
    
    __add = function(a, b)
        local result = {}
        for k, v in pairs(a) do
            result[k] = v
        end
        for k, v in pairs(b) do
            result[k] = v
        end
        return setmetatable(result, mt)
    end,
    
    __tostring = function(table)
        local result = "{"
        for k, v in pairs(table) do
            result = result .. k .. " = " .. tostring(v) .. ", "
        end
        return result .. "}"
    end
}

local myTable = setmetatable({}, mt)
myTable.x = 10
print(myTable.nonexistent) -- "Key not found: nonexistent"

        instance = {

            level = 1,
            players = {}
        }
        setmetatable(instance, {__index = GameManager})

end

Observer Pattern

local EventSystem = {}
EventSystem.__index = EventSystem

function EventSystem.new()
    return setmetatable({
        listeners = {}
    }, EventSystem)
end

function EventSystem:on(event, callback)
    if not self.listeners[event] then
        self.listeners[event] = {}
    end
    table.insert(self.listeners[event], callback)
end

function EventSystem:emit(event, ...)
    if self.listeners[event] then
        for i, callback in pairs(self.listeners[event]) do
            callback(...)
        end
    end
end

State Machine

local StateMachine = {}
StateMachine.__index = StateMachine

function StateMachine.new(initialState)
    return setmetatable({
        currentState = initialState,
        states = {},
        transitions = {}
    }, StateMachine)
end

function StateMachine:addState(name, enterFunc, updateFunc, exitFunc)
    self.states[name] = {
        enter = enterFunc,
        update = updateFunc,
        exit = exitFunc
    }
end

function StateMachine:transition(newState)
    local current = self.states[self.currentState]
    local next = self.states[newState]
    
    if current and current.exit then
        current.exit()
    end
    
    self.currentState = newState
    
    if next and next.enter then
        next.enter()
    end
end

function StateMachine:update(dt)
    local current = self.states[self.currentState]
    if current and current.update then
        current.update(dt)
    end
end

üí° Performance Tips:
  • Use local variables whenever possible
  • Cache function calls and properties
  • Avoid creating objects in loops
  • Use RunService for game loops
  • Disconnect unused connections

Object Pooling

local ObjectPool = {}
ObjectPool.__index = ObjectPool

function ObjectPool.new(createFunc, resetFunc)
    return setmetatable({
        pool = {},
        active = {},
        createFunc = createFunc,
        resetFunc = resetFunc
    }, ObjectPool)
end

function ObjectPool:get()
    local obj
    if #self.pool > 0 then
        obj = table.remove(self.pool)
    else
        obj = self.createFunc()
    end
    
    self.active[obj] = true
    return obj
end

function ObjectPool:return(obj)
    if self.active[obj] then
        self.resetFunc(obj)
        self.active[obj] = nil
        table.insert(self.pool, obj)
    end
end

Memory Management

-- Weak references
local weakTable = setmetatable({}, {__mode = "v"})

-- Cleanup function
local function cleanup()
    -- Disconnect all connections
    for i, connection in pairs(connections) do
        connection:Disconnect()
    end
    connections = {}
    
    -- Clear references
    for key in pairs(cache) do
        cache[key] = nil
    end
    
    -- Force garbage collection
    collectgarbage("collect")
end

-- Debounce function
local function debounce(func, delay)
    local timer = nil
    return function(...)
        local args = {...}
        if timer then
            timer:Disconnect()
        end
        timer = wait(delay)
        func(unpack(args))
    end
end
